#+title: Tmux
#+PROPERTY: header-args :tangle tmux.conf

* Tmux config
** Windows settings
#+BEGIN_SRC conf
# 0 is too far from ` ;)
set -g base-index 1
# Automatically set window title
set-window-option -g automatic-rename on
set-option -g set-titles on
#+END_SRC
** Vim keybindings
#+BEGIN_SRC conf

#set -g default-terminal screen-256color
set -g status-keys vi
set -g history-limit 10000

setw -g mode-keys vi
#setw -g mouse on
setw -g monitor-activity on
#+END_SRC
** Split windows
#+BEGIN_SRC conf
bind-key v split-window -h
bind-key s split-window -v
#+END_SRC
** Resize panes
*** Resize panes by 1 pixel
#+BEGIN_SRC conf
bind-key M-j resize-pane -D
bind-key M-k resize-pane -U
bind-key M-h resize-pane -L
bind-key M-l resize-pane -R
#+END_SRC
*** Resize panes by 5 pixels
#+BEGIN_SRC conf

bind-key J resize-pane -D 5
bind-key K resize-pane -U 5
bind-key H resize-pane -L 5
bind-key L resize-pane -R 5
#+END_SRC
** Select panes
*** Vim style pane selection
#+BEGIN_SRC conf
bind h select-pane -L
bind j select-pane -D
bind k select-pane -U
bind l select-pane -R
#+END_SRC
*** Use Alt+vim keys without prefix to swith panes
#+BEGIN_SRC conf
bind -n M-h select-pane -L
bind -n M-j select-pane -D
bind -n M-k select-pane -U
bind -n M-l select-pane -R
#+END_SRC
** Theme config
#+BEGIN_SRC conf
set -g status-bg black
set -g status-fg white
set -g status-interval 1
set -g status-left-length 30
set -g status-left '#[fg=green](#S) #(whoami) '
set -g status-right '#[default] #[fg=white]%H:%M:%S#[default]'
#+END_SRC

** Random settings
*** Reload tmux config with r
#+BEGIN_SRC conf
bind r source-file ~/.tmux.conf
#+END_SRC

*** No delay for escape key press
#+BEGIN_SRC conf
set -sg escape-time 0
#+END_SRC
